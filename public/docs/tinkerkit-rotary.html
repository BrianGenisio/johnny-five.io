Tinkerkit Rotary

Run with:

node src/j5/eg/tinkerkit-rotary.js

var five = require("johnny-five");

new five.Board().on("ready", function() {
  var servo = new five.Servo("O0");

  new five.Sensor("I1").scale(0, 180).on("change", function() {
    servo.to(this.value);
  });
});
