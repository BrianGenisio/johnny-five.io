Relay

Run with:

node src/j5/eg/relay.js

var five = require("johnny-five"),
  board = new five.Board();

board.on("ready", function() {
  var relay = new five.Relay(process.argv[2] || 10);

  this.repl.inject({
    relay: relay
  });
});
