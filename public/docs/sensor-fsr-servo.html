Sensor Fsr Servo

Run with:

node src/j5/eg/sensor-fsr-servo.js

var five = require("johnny-five"),
  board, fsr, servo;

board = new five.Board();

board.on("ready", function() {

  // Create a new `fsr` hardware instance.
  fsr = new five.Sensor({
    pin: "A0",
    freq: 25
  });

  servo = new five.Servo(10);

  fsr.scale([0, 180]).on("data", function() {

    servo.to(this.value);

  });
});
